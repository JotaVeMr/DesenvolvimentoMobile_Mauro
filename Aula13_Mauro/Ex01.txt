import 'package:flutter/material.dart';
import 'package:geolocator/geolocator.dart';

void main() {
  runApp(const MainApp());
}

// Classe que representa a localização
class Localizacao {
  double? latitude;
  double? longitude;

  // Método que obtém a localização atual
  Future<void> pegaLocalizacaoAtual() async {
    // Solicita permissão
    LocationPermission permission = await Geolocator.requestPermission();

    if (permission == LocationPermission.denied ||
        permission == LocationPermission.deniedForever) {
      throw Exception("Permissão de localização negada");
    }

    // Obtém a posição atual
    Position posicao = await Geolocator.getCurrentPosition(
      desiredAccuracy: LocationAccuracy.high,
    );

    latitude = posicao.latitude;
    longitude = posicao.longitude;
  }
}

class MainApp extends StatefulWidget {
  const MainApp({super.key});

  @override
  State<MainApp> createState() => _MainAppState();
}

class _MainAppState extends State<MainApp> {
  Localizacao localizacao = Localizacao();
  String latitudeTexto = "?";
  String longitudeTexto = "?";

  @override
  void initState() {
    super.initState();
    _obterLocalizacao();
  }

  Future<void> _obterLocalizacao() async {
    try {
      await localizacao.pegaLocalizacaoAtual();
      setState(() {
        latitudeTexto = localizacao.latitude?.toStringAsFixed(6) ?? "?";
        longitudeTexto = localizacao.longitude?.toStringAsFixed(6) ?? "?";
      });
    } catch (e) {
      setState(() {
        latitudeTexto = "Erro";
        longitudeTexto = "Erro";
      });
    }
  }

  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      home: Scaffold(
        appBar: AppBar(title: const Text("Busca Local")),
        body: Center(
          child: Column(
            mainAxisAlignment: MainAxisAlignment.center,
            children: [
              const Text(
                'Busca Local',
                style: TextStyle(fontSize: 24, fontWeight: FontWeight.bold),
              ),
              const SizedBox(height: 20),
              Text('Latitude: $latitudeTexto'),
              Text('Longitude: $longitudeTexto'),
              const SizedBox(height: 20),
              ElevatedButton(
                onPressed: _obterLocalizacao,
                child: const Text("Atualizar Localização"),
              ),
            ],
          ),
        ),
      ),
    );
  }
}
