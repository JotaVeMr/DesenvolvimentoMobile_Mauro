import 'dart:convert';
import 'package:flutter/material.dart';
import 'package:http/http.dart' as http;

void main() {
  runApp(MaterialApp(
    home: DictionaryPage(),
    debugShowCheckedModeBanner: false,
  ));
}

class DictionaryPage extends StatefulWidget {
  @override
  State<DictionaryPage> createState() => _DictionaryPageState();
}

class _DictionaryPageState extends State<DictionaryPage> {
  final TextEditingController _controller = TextEditingController();

  String? primeiraDefinicao;
  bool loading = false;
  String? erro;

  Future<void> buscarDefinicao(String palavra) async {
    setState(() {
      loading = true;
      erro = null;
      primeiraDefinicao = null;
    });

    try {
      final url = "https://api.dictionaryapi.dev/api/v2/entries/en/$palavra";
      final response = await http.get(Uri.parse(url));

      if (response.statusCode != 200) {
        setState(() {
          erro = "Palavra nÃ£o encontrada.";
          loading = false;
        });
        return;
      }

      final data = json.decode(response.body);

      
      final definicao = data[0]["meanings"][0]["definitions"][0]["definition"];

      setState(() {
        primeiraDefinicao = definicao;
        loading = false;
      });
    } catch (e) {
      setState(() {
        erro = "Erro ao buscar definiÃ§Ã£o.";
        loading = false;
      });
    }
  }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(
        title: Text("Mini DicionÃ¡rio ðŸ‡ºðŸ‡¸"),
        centerTitle: true,
      ),
      body: Padding(
        padding: const EdgeInsets.all(16),
        child: Column(
          children: [
            TextField(
              controller: _controller,
              decoration: InputDecoration(
                labelText: "Digite uma palavra em inglÃªs",
                border: OutlineInputBorder(),
              ),
            ),
            const SizedBox(height: 12),
            ElevatedButton(
              onPressed: () {
                final palavra = _controller.text.trim();
                if (palavra.isNotEmpty) {
                  buscarDefinicao(palavra);
                }
              },
              child: Text("Buscar DefiniÃ§Ã£o"),
            ),
            const SizedBox(height: 20),

            if (loading) CircularProgressIndicator(),

            if (erro != null)
              Text(erro!, style: TextStyle(color: Colors.red, fontSize: 16)),

            if (primeiraDefinicao != null)
              Column(
                crossAxisAlignment: CrossAxisAlignment.start,
                children: [
                  Text(
                    "DefiniÃ§Ã£o:",
                    style: TextStyle(
                        fontSize: 20,
                        fontWeight: FontWeight.bold
                    ),
                  ),
                  const SizedBox(height: 10),
                  Text(
                    primeiraDefinicao!,
                    style: TextStyle(fontSize: 18),
                  ),
                ],
              ),
          ],
        ),
      ),
    );
  }
}
